Inputs:
	GYear(Numeric),
	GMonth(Numeric),
	GDay(Numeric);

Variables:
	RFactor(0);

If GMonth <= 2 Then
	RFactor = 0
Else If GMonth > 2 and sfgf.gleapyear(GYear) Then
	RFactor = -1
Else
	RFactor = -2;

sfgf.gdate2rdate = 365 * (GYear - 1) +
					  Floor((GYear - 1) / 4) -
					  Floor((GYear - 1) / 100) +
					  Floor((GYear - 1) / 400) +
					  Floor(((367 * GMonth) - 362) / 12) +
					  GDay + RFactor;
